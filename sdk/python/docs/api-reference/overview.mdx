
## Import Reference

```python
# Main clients
from superserve_sdk import Superserve, SuperserveSync

# Models
from superserve_sdk import Agent, Run, UsageMetrics

# Events
from superserve_sdk import (
    RunEvent,
    RunStartedEvent,
    MessageDeltaEvent,
    ToolStartEvent,
    ToolEndEvent,
    RunCompletedEvent,
    RunFailedEvent,
    RunCancelledEvent,
)

# Metrics
from superserve_sdk import RunMetrics, ToolMetrics, MetricsCollector

# Errors
from superserve_sdk import (
    SuperserveError,
    APIError,
    AuthenticationError,
    NotFoundError,
    ValidationError,
    ConflictError,
    RateLimitError,
    StreamError,
    ConnectionError,
    TimeoutError,
)

# Authentication
from superserve_sdk import (
    AuthProvider,
    get_api_key,
    save_credentials_to_file,
    load_credentials_from_file,
    clear_credentials,
)
```

## Type Annotations

The SDK is fully typed and works with mypy and other type checkers:

```python
from superserve_sdk import Superserve, Agent, Run

async def example(client: Superserve) -> Run:
    agent: Agent = await client.create_agent(name="typed-agent")
    run: Run = await client.run(agent.id, "Hello")
    return run
```
